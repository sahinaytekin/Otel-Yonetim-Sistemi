CREATE DATABASE HOTELMANAGEMENT;

CREATE TABLE GUEST(
    GUESTNAME VARCHAR(30),
    GENDER VARCHAR(30),
    STATUS VARCHAR(30),
    CITY VARCHAR(30),
    COUNTRY VARCHAR(30),
    PHONE VARCHAR(30),
    ROOMNO INT PRIMARY KEY,
    NIGHTSTAY INT,
    CHECKIN VARCHAR(30),
    CHECKOUT VARCHAR(30)
    );

CREATE TABLE ROOM( 
	ROOMNO INT PRIMARY KEY, 
	ROOMTYPE VARCHAR(30), 
	BEDTYPE VARCHAR(30), 
	TARIFF DOUBLE, 
	STATUS BOOLEAN, 
	GUEST VARCHAR(30) 
	);

CREATE TABLE MANAGER(
    	NAME VARCHAR(30),
        ADDRESS VARCHAR(30),
    	PHONE VARCHAR(30),
    	EMAIL VARCHAR(30), 
    	PASSWORD VARCHAR(30)
    );
